237  [[nodiscard]] static Self
238  slerp(const Self& qa, const float t, const Self& qb)
239  {
240  // from:
241  // http://www.euclideanspace.com/maths/algebra/realNormedAlgebra/quaternions/slerp/
242  // Calculate angle between them.
243  const float cos_half_theta = qa.w * qb.w + qa.x * qb.x + qa.y * qb.y + qa.z * qb.z;
244  // if qa=qb or qa=-qb then theta = 0 and we can return qa
245  if(abs(cos_half_theta) >= 1.0f)
246  {
247  return qa;
248  }
249  // Calculate temporary values.
250  const auto half_theta = acos(cos_half_theta);
251  const auto sin_half_theta = sqrt(1.0f - cos_half_theta * cos_half_theta);
252  if(abs(sin_half_theta) < 0.001f)
253  {
254  // if theta = 180 degrees then result is not fully defined
255  // we could rotate around any axis normal to qa or qb
256  const Self qt = qa + qb;
257  return Self
258  {
259  qt.w * 0.5f,
260  Vec
261  {
262  qt.x * 0.5f,
263  qt.y * 0.5f,
264  qt.z * 0.5f
265  }
266  };
267  }
268  const float ratio_a = sin((1 - t) * half_theta) / sin_half_theta;
269  const float ratio_b = sin(t * half_theta) / sin_half_theta;
270  return qa * ratio_a + qb * ratio_b;
271  }
