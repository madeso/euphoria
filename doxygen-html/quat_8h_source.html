289  [[nodiscard]] static Self
290  slerp(const Self& qa, const float t, const Self& qb)
291  {
292  // from:
293  // http://www.euclideanspace.com/maths/algebra/realNormedAlgebra/quaternions/slerp/
294  // Calculate angle between them.
295  const float cos_half_theta = qa.w * qb.w + qa.x * qb.x + qa.y * qb.y + qa.z * qb.z;
296  // if qa=qb or qa=-qb then theta = 0 and we can return qa
297  if(abs(cos_half_theta) >= 1.0f)
298  {
299  return qa;
300  }
301  // Calculate temporary values.
302  const auto half_theta = acos(cos_half_theta);
303  const auto sin_half_theta = sqrt(1.0f - cos_half_theta * cos_half_theta);
304  if(abs(sin_half_theta) < 0.001f)
305  {
306  // if theta = 180 degrees then result is not fully defined
307  // we could rotate around any axis normal to qa or qb
308  const Self qt = qa + qb;
309  return Self
310  {
311  qt.w * 0.5f,
312  Vec
313  {
314  qt.x * 0.5f,
315  qt.y * 0.5f,
316  qt.z * 0.5f
317  }
318  };
319  }
320  const float ratio_a = sin((1 - t) * half_theta) / sin_half_theta;
321  const float ratio_b = sin(t * half_theta) / sin_half_theta;
322  return qa * ratio_a + qb * ratio_b;
323  }
