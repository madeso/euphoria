109  std::optional<float>
110  calc_zoom_move(int dx, int dy, float length, EditorCamera3* owner)
111  {
112  const auto change = dx+dy;
113  if (change == 0) { return std::nullopt; }
114 
115  const auto suggested_move = c_int_to_float(change) * length * owner->zoom_percent;
116  const auto sign = core::get_sign(change > 0);
117  const auto val = core::min(core::square(suggested_move), length);
118  return sign * core::min(val, owner->max_zoom_change);
119  }
