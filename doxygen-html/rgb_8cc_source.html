255  crgb(const Hsl& hsl)
256  {
257  // based on https://gist.github.com/mjackson/5311256
258  if(hsl.s == 0)
259  {
260  return rgb {hsl.l}; // achromatic
261  }
262  else
263  {
264  auto hue2rgb = [](float p, float q, float t) {
265  if(t < 0.0f) { t += 1.0f; }
266  if(t > 1.0f) { t -= 1.0f; }
267 
268  if(t < 1.0f / 6.0f) { return p + (q - p) * 6.0f * t; }
269  else if(t < 1.0f / 2.0f) { return q; }
270  else if(t < 2.0f / 3.0f) { return p + (q - p) * (2.0f / 3.0f - t) * 6.0f; }
271  else { return p; }
272  };

275  : hsl.l + hsl.s - hsl.l * hsl.s;
276  const auto p = 2.0f * hsl.l - q;
277 
278  const auto r = hue2rgb(p, q, hsl.h.in_percent_of_360() + 1.0f / 3.0f);
279  const auto g = hue2rgb(p, q, hsl.h.in_percent_of_360());
280  const auto b = hue2rgb(p, q, hsl.h.in_percent_of_360() - 1.0f / 3.0f);
