34 quatf::from_ypr(const Angle& yaw, const Angle& pitch, const Angle& roll)
35 {
36 // Abbreviations for the various angular functions
37 const auto cy = cos(yaw * 0.5);
38 const auto sy = sin(yaw * 0.5);
39 const auto cp = cos(pitch * 0.5);
40 const auto sp = sin(pitch * 0.5);
41 const auto cr = cos(roll * 0.5);
42 const auto sr = sin(roll * 0.5);
43
44 return
45 {
46 cr * cp * cy + sr * sp * sy,
47 {
48 sr * cp * cy - cr * sp * sy,
49 cr * sp * cy + sr * cp * sy,
50 cr * cp * sy - sr * sp * cy
51 }
52 };

210 quatf
211 slerp_fast(const quatf& qa, const float t, const quatf& qb)
212 {
213 // from:
214 // http://www.euclideanspace.com/maths/algebra/realNormedAlgebra/quaternions/slerp/
215 // Calculate angle between them.
216 const float cos_half_theta = qa.w * qb.w + qa.x * qb.x + qa.y * qb.y + qa.z * qb.z;
217 // if qa=qb or qa=-qb then theta = 0 and we can return qa
218 if(abs(cos_half_theta) >= 1.0f)
219 {
220 return qa;
221 }
222 // Calculate temporary values.
223 const auto half_theta = acos(cos_half_theta);
224 const auto sin_half_theta = sqrt(1.0f - cos_half_theta * cos_half_theta);
225 if(abs(sin_half_theta) < 0.001f)
226 {
227 // if theta = 180 degrees then result is not fully defined
228 // we could rotate around any axis normal to qa or qb
229 const quatf qt = qa + qb;
230 return quatf
231 {
232 qt.w * 0.5f,
233 v3
234 {
235 qt.x * 0.5f,
236 qt.y * 0.5f,
237 qt.z * 0.5f
238 }
239 };
240 }
241 const float ratio_a = sin((1 - t) * half_theta) / sin_half_theta;
242 const float ratio_b = sin(t * half_theta) / sin_half_theta;
243 return qa * ratio_a + qb * ratio_b;
244 }
