34 Q::from(const Ypr& ypr)
35 {
36 // Abbreviations for the various angular functions
37 const auto cy = cos(ypr.yaw * 0.5);
38 const auto sy = sin(ypr.yaw * 0.5);
39 const auto cp = cos(ypr.pitch * 0.5);
40 const auto sp = sin(ypr.pitch * 0.5);
41 const auto cr = cos(ypr.roll * 0.5);
42 const auto sr = sin(ypr.roll * 0.5);
43
44 return
45 {
46 cr * cp * cy + sr * sp * sy,
47 {
48 sr * cp * cy - cr * sp * sy,
49 cr * sp * cy + sr * cp * sy,
50 cr * cp * sy - sr * sp * cy
51 }
52 };

187 Q
188 slerp_fast(const Q& qa, const float t, const Q& qb)
189 {
190 // from:
191 // http://www.euclideanspace.com/maths/algebra/realNormedAlgebra/quaternions/slerp/
192 // Calculate angle between them.
193 const float cos_half_theta = qa.w * qb.w + qa.x * qb.x + qa.y * qb.y + qa.z * qb.z;
194 // if qa=qb or qa=-qb then theta = 0 and we can return qa
195 if(abs(cos_half_theta) >= 1.0f)
196 {
197 return qa;
198 }
199 // Calculate temporary values.
200 const auto half_theta = acos(cos_half_theta);
201 const auto sin_half_theta = sqrt(1.0f - cos_half_theta * cos_half_theta);
202 if(abs(sin_half_theta) < 0.001f)
203 {
204 // if theta = 180 degrees then result is not fully defined
205 // we could rotate around any axis normal to qa or qb
206 const Q qt = qa + qb;
207 return Q
208 {
209 qt.w * 0.5f,
210 v3
211 {
212 qt.x * 0.5f,
213 qt.y * 0.5f,
214 qt.z * 0.5f
215 }
216 };
217 }
218 const float ratio_a = sin((1 - t) * half_theta) / sin_half_theta;
219 const float ratio_b = sin(t * half_theta) / sin_half_theta;
220 return qa * ratio_a + qb * ratio_b;
221 }
