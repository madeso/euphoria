261  crgb(const Hsl& hsl)
262  {
263  // based on https://gist.github.com/mjackson/5311256
264  if(hsl.s == 0)
265  {
266  return Rgb {hsl.l}; // achromatic
267  }
268  else
269  {
270  auto hue2rgb = [](float p, float q, float t) {
271  if(t < 0.0f) { t += 1.0f; }
272  if(t > 1.0f) { t -= 1.0f; }
273 
274  if(t < 1.0f / 6.0f) { return p + (q - p) * 6.0f * t; }
275  else if(t < 1.0f / 2.0f) { return q; }
276  else if(t < 2.0f / 3.0f) { return p + (q - p) * (2.0f / 3.0f - t) * 6.0f; }
277  else { return p; }
278  };
279 
280  const auto q = hsl.l < 0.5f ? hsl.l * (1.0f + hsl.s)
281  : hsl.l + hsl.s - hsl.l * hsl.s;
282  const auto p = 2.0f * hsl.l - q;
283 
284  const auto r = hue2rgb(p, q, hsl.h.in_percent_of_360() + 1.0f / 3.0f);
285  const auto g = hue2rgb(p, q, hsl.h.in_percent_of_360());
286  const auto b = hue2rgb(p, q, hsl.h.in_percent_of_360() - 1.0f / 3.0f);
287  return {r, g, b};
288  }

303  {
304  r,
305  g,
306  b,
307  same
308  };
309 
310  const auto cl = [](float r, float g, float b) -> BiggestValue {
311  constexpr auto min_diff = 0.001f;
312  if(abs(r - g) < min_diff && abs(g - b) < min_diff)
313  {
314  return BiggestValue::same;
315  }
316  if(r >= g && r >= b)
317  {
318  return BiggestValue::r;
319  }
320  if(g >= r && g >= b)
321  {
322  return BiggestValue::g;
323  }
324  ASSERTX(b >= r && b >= g, r, g, b);
325  return BiggestValue::b;
326  }(c.r, c.g, c.b);
327 
328  if(cl == BiggestValue::same)
329  {
330  return {Angle::from_radians(0), 0, l}; // achromatic
331  }
332  else

335  const auto s = l > 0.5f ? d / (2 - max - min) : d / (max + min);
336

338  {
339  switch(cl)
340  {
341  case BiggestValue::r: return (c.g - c.b) / d + (c.g < c.b ? 6.0f : 0.0f);
342  case BiggestValue::g: return (c.b - c.r) / d + 2;
343  case BiggestValue::b: return (c.r - c.g) / d + 4;
344  default: DIE("Unreachable"); return 0.0f;
